import Dexie, { type Table } from 'dexie'

// DB schema definition
class OfflineDatabase extends Dexie {
  offlineTickets!: Table<OfflineTicket>
  mutationQueue!: Table<MutationQueueEntry>
  cachedAssets!: Table<CachedAsset>

  constructor() {
    super('lionheart-offline')
    this.version(1).stores({
      offlineTickets: '++id, ticketId, organizationId, assignedToId, status, updatedAt',
      mutationQueue: '++id, type, ticketId, status, createdAt',
      cachedAssets: '++id, assetId, assetNumber, organizationId',
    })
  }
}
export const db = new OfflineDatabase()

// Usage
await db.mutationQueue.add({ type: 'STATUS_UPDATE', ticketId, payload: { status: 'IN_PROGRESS' }, createdAt: new Date() })
await db.mutationQueue.orderBy('createdAt').toArray()
await db.mutationQueue.delete(id)

     id?: number           // Dexie auto-increment PK

     organizationId: string
     assignedToId: string

     assetId?: string

   }

   export interface MutationQueueEntry {
     id?: number           // Dexie auto-increment PK
     type: 'TICKET_CREATE' | 'STATUS_UPDATE' | 'LABOR_LOG' | 'COST_LOG' | 'CHECKLIST_TOGGLE' | 'COMMENT_ADD'
     ticketId: string      // Server ID or temp ID (temp IDs start with 'temp-')
     payload: Record<string, unknown>
     status: 'pending' | 'syncing' | 'failed'
     retryCount: number
     createdAt: Date
     error?: string
   }

   export interface CachedAsset {
     id?: number
     assetId: string
     organizationId: string
     assetNumber: string
     name: string
     category?: string
     buildingId?: string
     areaId?: string
     roomId?: string
     cachedAt: Date
   }

   class OfflineDatabase extends Dexie {
     offlineTickets!: Table<OfflineTicket>
     mutationQueue!: Table<MutationQueueEntry>
     cachedAssets!: Table<CachedAsset>

     constructor() {
       super('lionheart-offline-v1')
       this.version(1).stores({
         offlineTickets: '++id, ticketId, organizationId, assignedToId, status, cachedAt',
         mutationQueue: '++id, type, ticketId, status, createdAt',
         cachedAssets: '++id, assetId, organizationId, assetNumber',
       })
     }
   }

   export const db = new OfflineDatabase()
   ```

3. **src/lib/offline/queue.ts** — Queue and drain operations:
   - `enqueueOfflineMutation(entry: Omit<MutationQueueEntry, 'id' | 'status' | 'retryCount'>)`: Adds to queue with status='pending', retryCount=0
   - `getPendingMutations()`: Returns all queue entries ordered by createdAt asc
   - `markMutationSyncing(id: number)`: Set status='syncing'
   - `markMutationFailed(id: number, error: string)`: Set status='failed', increment retryCount, store error
   - `deleteMutation(id: number)`: Remove after successful sync
   - `getQueueCount()`: Returns count of pending+syncing+failed entries
   - All functions are simple Dexie table operations

4. **src/lib/offline/conflicts.ts** — Conflict resolution strategies:
   - `resolveStatusConflict(localStatus: string, serverStatus: string, localUpdatedAt: Date, serverUpdatedAt: Date): string`:
     Last-write-wins: return `localUpdatedAt > serverUpdatedAt ? localStatus : serverStatus`
   - `resolveCommentConflict(localComments: string[], serverComments: string[]): string[]`:
     Merge (append): return `[...serverComments, ...localComments.filter(c => !serverComments.includes(c))]`
     This is a simplified merge — in practice comments have unique IDs so dedup is by content. The server's activity feed is the source of truth; local comments are appended if not already present.

5. **src/lib/offline/sync.ts** — Background sync orchestrator:
   - `cacheAssignedTickets(authToken: string, organizationId: string, userId: string)`:
     - Fetch `GET /api/maintenance/tickets?assignedTo=me` with auth headers
     - On success: upsert all returned tickets into `db.offlineTickets` (update existing by ticketId, insert new ones). Set `isLocalOnly: false`, `cachedAt: new Date()`.
     - Also fetch `GET /api/maintenance/assets` and upsert into `db.cachedAssets`
     - Called on login/app load when online

   - `syncOfflineData(queryClient: QueryClient, authToken: string)`:
     - Get all pending mutations from queue (ordered by createdAt)
     - Process each mutation sequentially (not in parallel — order matters for status transitions):
       ```
       for mutation in queue:
         markMutationSyncing(mutation.id)
         try:
           result = await applyMutation(mutation, authToken)
           await deleteMutation(mutation.id)
           if mutation.type === 'TICKET_CREATE' && result.newTicketId:
             // Update any subsequent mutations that reference the temp ID
             await updateTempIdReferences(mutation.ticketId, result.newTicketId)
         catch error:
           await markMutationFailed(mutation.id, error.message)
       ```
     - After processing all: call `queryClient.invalidateQueries({ queryKey: ['maintenance-tickets'] })` and `queryClient.invalidateQueries({ queryKey: ['maintenance-assets'] })`

   - `applyMutation(mutation: MutationQueueEntry, authToken: string): Promise<{ newTicketId?: string }>`:
     Route by type:
     - `TICKET_CREATE`: POST /api/maintenance/tickets with payload. Return `{ newTicketId: result.data.id }`
     - `STATUS_UPDATE`: PATCH /api/maintenance/tickets/[ticketId] with `{ status, ... }`. Ignore 409 conflicts (server wins on concurrent update). Apply last-write-wins if server returns 200 with a different status.
     - `LABOR_LOG`: POST /api/maintenance/tickets/[ticketId]/labor with payload
     - `COST_LOG`: POST /api/maintenance/tickets/[ticketId]/costs with payload
     - `CHECKLIST_TOGGLE`: PATCH /api/maintenance/tickets/[ticketId]/checklist with `{ index, done }`
     - `COMMENT_ADD`: POST /api/maintenance/tickets/[ticketId]/activities with `{ type: 'COMMENT', content }`
     All requests include `Authorization: Bearer {authToken}` header

6. **src/hooks/useConnectivity.ts**:

   'use client'
   import { useState, useEffect } from 'react'
   export function useConnectivity() {
     const [isOnline, setIsOnline] = useState(() => typeof navigator !== 'undefined' ? navigator.onLine : true)
     useEffect(() => {
       const handleOnline = () => setIsOnline(true)
       const handleOffline = () => setIsOnline(false)
       window.addEventListener('online', handleOnline)
       window.addEventListener('offline', handleOffline)
       return () => {
         window.removeEventListener('online', handleOnline)
         window.removeEventListener('offline', handleOffline)
       }
     }, [])
     return isOnline
   }
   ```

**1. src/lib/services/registrationPaymentService.ts** — Stripe payment management.

```typescript
import Stripe from 'stripe'
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!)
```

Functions:
- `createPaymentIntent(data: { registrationId, amount: number, paymentType: 'FULL' | 'DEPOSIT' | 'BALANCE', discountCode?: string, eventProjectId, organizationId })` —
  1. If discountCode provided, call applyDiscountCode to compute discounted amount
  2. Create Stripe PaymentIntent with `automatic_payment_methods: { enabled: true }`, metadata: { registrationId, eventProjectId, organizationId, paymentType, discountCode }
  3. Create RegistrationPayment row with stripePaymentIntentId, amount, status 'requires_action', paymentType
  4. Return { clientSecret, paymentIntentId, amount }

- `createBalanceIntent(registrationId: string)` — Creates a PaymentIntent for the remaining balance after a deposit was paid.
  1. Fetch the registration + form to get basePrice and depositPercent
  2. Fetch existing RegistrationPayments for this registration to calculate already-paid amount
  3. Compute balanceAmount = basePrice - totalPaid (in cents)
  4. If balanceAmount <= 0, throw "No balance remaining"
  5. Call createPaymentIntent with paymentType 'BALANCE' and the computed balanceAmount
  6. Return { clientSecret, paymentIntentId, amount: balanceAmount }

- `applyDiscountCode(baseAmount: number, discountCode: string, formId: string)` — Looks up discount code in RegistrationForm.discountCodes JSON. Validates code exists, hasn't exceeded maxUses. Returns { discountedAmount, discountAmount }. Throws if code invalid.

- `handlePaymentSuccess(stripePaymentIntentId: string)` — Called from webhook. Upserts RegistrationPayment with status 'succeeded', paidAt. Updates EventRegistration.paymentStatus to DEPOSIT_PAID or PAID based on paymentType. For BALANCE paymentType, sets paymentStatus to PAID. Uses rawPrisma.

- `calculateAmount(form: RegistrationForm, paymentType: 'FULL' | 'DEPOSIT')` — Returns amount in cents. DEPOSIT = basePrice * depositPercent / 100. FULL = basePrice.

```typescript
/**
 * Upload a student registration photo to Supabase Storage.
 * Uses a private 'student-photos' bucket with signed URLs for viewing.
 *
 * @param orgId - Organization ID
 * @param registrationId - Registration ID (used in path)
 * @param fileBuffer - File buffer to upload
 * @param contentType - MIME type (e.g., 'image/jpeg')
 * @returns Public URL of the uploaded file
 */
export async function uploadRegistrationPhoto(
  orgId: string,
  registrationId: string,
  fileBuffer: Buffer,
  contentType: string
): Promise<string> {
  const config = getStorageConfig()
  if (!config) throw new Error('Supabase storage not configured')

  const client = getSupabaseClient()
  const ext = contentType.split('/')[1] || 'jpg'
  const fileName = `${Date.now()}-${Math.random().toString(36).slice(2, 8)}.${ext}`
  const path = `${orgId}/${registrationId}/${fileName}`

  const { error } = await client.storage.from('student-photos').upload(path, fileBuffer, {
    contentType,
    upsert: false,
  })

  if (error) throw new Error(`Upload failed: ${error.message}`)

  // Return the public URL path (bucket is private — generate signed URLs for viewing)
  return `${config.url}/storage/v1/object/public/student-photos/${path}`
}
```

Also add a `getSignedPhotoUrl` function:
```typescript
export async function getSignedPhotoUrl(photoPath: string, expiresIn = 3600): Promise<string> {
  const client = getSupabaseClient()
  const { data, error } = await client.storage.from('student-photos').createSignedUrl(photoPath, expiresIn)
  if (error || !data) throw new Error(`Signed URL failed: ${error?.message}`)
  return data.signedUrl
}
```

Note: The 'student-photos' bucket should be created as PRIVATE in Supabase. Access is via signed URLs only (1-hour TTL for viewing). This resolves the RESEARCH.md open question about bucket name and access policy.

<!-- API routes from Plan 03 -->

Schedule Block API:
GET /api/events/projects/[id]/schedule — returns { ok: true, data: EventScheduleBlock[] }
POST /api/events/projects/[id]/schedule — body: { type, title, description?, startsAt, endsAt, locationText?, leadId?, sortOrder? } -> 201
PATCH /api/events/projects/[id]/schedule/[blockId] — body: partial update -> 200
DELETE /api/events/projects/[id]/schedule/[blockId] -> 200

Task API:
GET /api/events/projects/[id]/tasks — returns { ok: true, data: EventTask[] }
POST /api/events/projects/[id]/tasks — body: { title, description?, status?, priority?, category?, assigneeId?, dueDate? } -> 201
PATCH /api/events/projects/[id]/tasks/[taskId] — body: partial update -> 200
DELETE /api/events/projects/[id]/tasks/[taskId] -> 200

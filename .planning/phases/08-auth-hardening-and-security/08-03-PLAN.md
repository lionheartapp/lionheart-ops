<objective>
Migrate JWT storage from localStorage to httpOnly cookies with CSRF protection. This is the most pervasive change in phase 8 — it touches auth flow, middleware, API client, and the client auth hook.

Purpose: AUTH-04 (httpOnly cookies) and AUTH-05 (CSRF protection) are critical security requirements. localStorage tokens are accessible to XSS attacks; httpOnly cookies are not. CSRF protection prevents cross-site request forgery on state-changing endpoints.
Output: Cookie-based auth flow, /api/auth/me endpoint, CSRF double-submit cookie pattern, migrated api-client.ts and useAuth hook.
</objective>

   export async function fetchApi<T>(url: string, options?: RequestInit): Promise<T> {
     const res = await fetch(url, {
       ...options,
       credentials: 'include',
       headers: { ...getAuthHeaders(), ...options?.headers },
     })
     if (res.status === 401 && typeof window !== 'undefined') {
       // Clean up any legacy localStorage data

     ```typescript
     export function useAuth({ redirectTo = '/login' } = {}): AuthState {
       const router = useRouter()
       const [isReady, setIsReady] = useState(false)
       const [user, setUser] = useState<AuthUser>({ name: 'User', email: '', avatar: null, team: null, schoolScope: null, role: null })
       const [org, setOrg] = useState<AuthOrg>({ id: '', name: 'School', schoolType: null, logoUrl: null })
       const [isAuthenticated, setIsAuthenticated] = useState(false)

       useEffect(() => {
         let cancelled = false
         fetch('/api/auth/me', { credentials: 'include' })
           .then(res => {
             if (!res.ok) throw new Error('Not authenticated')
             return res.json()
           })
           .then(json => {
             if (cancelled) return
             if (!json.ok) throw new Error('Not authenticated')
             setUser(json.data.user)
             setOrg(json.data.org)
             setIsAuthenticated(true)
             setIsReady(true)
           })
           .catch(() => {
             if (cancelled) return
             setIsReady(true)
             if (redirectTo) router.push(redirectTo)
           })
         return () => { cancelled = true }
       }, [redirectTo, router])

       // Keep avatar/profile update listeners
       useEffect(() => { /* same event listener logic for avatar-updated and profile-updated */ }, [])

       const logout = useCallback(async () => {
         await fetch('/api/auth/logout', { method: 'POST', credentials: 'include' })
         const legacyKeys = ['auth-token', 'org-id', 'user-name', 'user-email', 'user-avatar', 'user-team', 'user-school-scope', 'user-role', 'org-name', 'org-school-type', 'org-logo-url']
         legacyKeys.forEach(k => localStorage.removeItem(k))
         router.push('/login')
       }, [router])

       const isAdmin = useMemo(() => {
         const role = (user.role || '').toLowerCase()
         return role.includes('admin') || role.includes('super')
       }, [user.role])

       return { token: null, orgId: org.id || null, user, org, isReady, isAdmin, logout }
     }
     ```
